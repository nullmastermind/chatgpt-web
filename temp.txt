function findHighlight(text: string, highlight: string): string[] {
  const result: string[] = [];
  const highlightChars = highlight.split("");

  for (let i = 0; i < text.length; i++) {
    let j = 0;
    while (j < highlightChars.length && i + j < text.length && text[i + j] === highlightChars[j]) {
      j++;
    }
    if (j === highlightChars.length) {

  }

}
